const SHEET_ID = '1T6Ii49JP_DChJNqqUEnQAV0POmxPQ1DWkzML9xAYMhU'; function doPost(e) { try { const data = JSON.parse(e.parameter.payload); const ss = SpreadsheetApp.openById(SHEET_ID); // Foglio Profilo let shP = ss.getSheetByName('Profilo') || ss.insertSheet('Profilo'); shP.clear(); shP.getRange(1,1,1,2).setValues([['Chiave', 'Valore']]); let row = 2; for (let k in data.profilo) { shP.getRange(row, 1, 1, 2).setValues([[k, data.profilo[k]]]); row++; } // Foglio Scarpe let shS = ss.getSheetByName('Scarpe') || ss.insertSheet('Scarpe'); shS.clear(); shS.getRange(1,1,1,5).setValues([['Nome','Tipo','KmMax','KmPercorsi','Attiva']]); data.scarpe.forEach((s, i) => { shS.getRange(i+2, 1, 1, 5).setValues([[s.nome, s.tipo, s.kmMax, s.kmPercorsi, s.attiva]]); }); // Foglio Diario let shD = ss.getSheetByName('Diario') || ss.insertSheet('Diario'); shD.clear(); shD.getRange(1,1,1,7).setValues([['Data','Tipo','Km','Passo','FC','RPE','Scarpa']]); data.diario.forEach((w, i) => { shD.getRange(i+2, 1, 1, 7).setValues([[w.data, w.tipo, w.km, w.passo, w.fc, w.rpe, w.scarpa]]); }); // Timestamp sync let shLog = ss.getSheetByName('Log') || ss.insertSheet('Log'); shLog.getRange(1,1).setValue('Ultimo sync: ' + new Date()); return ContentService.createTextOutput('OK'); } catch(err) { return ContentService.createTextOutput('ERR: ' + err); } } function doGet(e) { const ss = SpreadsheetApp.openById(SHEET_ID); const profilo = {}; const shP = ss.getSheetByName('Profilo'); if (shP) { const vals = shP.getRange(2,1,shP.getLastRow()-1,2).getValues(); vals.forEach(r => profilo[r[0]] = r[1]); } const scarpe = []; const shS = ss.getSheetByName('Scarpe'); if (shS && shS.getLastRow() > 1) { const vals = shS.getRange(2,1,shS.getLastRow()-1,5).getValues(); vals.forEach(r => scarpe.push({nome:r[0], tipo:r[1], kmMax:r[2], kmPercorsi:r[3], attiva:r[4]})); } const diario = []; const shD = ss.getSheetByName('Diario'); if (shD && shD.getLastRow() > 1) { const vals = shD.getRange(2,1,shD.getLastRow()-1,7).getValues(); vals.forEach(r => diario.push({data:r[0], tipo:r[1], km:r[2], passo:r[3], fc:r[4], rpe:r[5], scarpa:r[6]})); } const result = {profilo, scarpe, diario}; return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); }
